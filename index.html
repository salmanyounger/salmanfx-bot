// ============================================================ // SalmanFX Bot Server — server.js v3 // Fixes: Real MT4 dates, Yahoo Finance gold price // Run: node server.js // ============================================================ const http = require("http"); const https = require("https"); const WebSocket = require("ws"); const PORT = 3008; const accounts = {}; const contacts = []; let goldPrice = null; let goldPrev = null; // ─── Parse MT4 date "2026.04.02 02:14" → JS Date ──────────── function parseMT4Date(str) { if (!str) return new Date(); try { const cleaned = str.replace(/\./g, "-").replace(" ", "T") + ":00"; const d = new Date(cleaned); return isNaN(d.getTime()) ? new Date() : d; } catch(e) { return new Date(); } } // ─── Yahoo Finance gold price (free, no API key) ───────────── function fetchGoldPrice() { const options = { hostname: "query1.finance.yahoo.com", path: "/v8/finance/chart/GC%3DF?interval=1m&range=1d", method: "GET", headers: { "User-Agent": "Mozilla/5.0", "Accept": "application/json" } }; const req = https.request(options, (res) => { let data = ""; res.on("data", chunk => data += chunk); res.on("end", () => { try { const json = JSON.parse(data); const price = json.chart.result[0].meta.regularMarketPrice; if (price && price > 100) { goldPrev = goldPrice; goldPrice = parseFloat(price.toFixed(2)); console.log(`[🪙] Gold: $${goldPrice}`); broadcast({ type:"gold_price", price:goldPrice, prev:goldPrev }); } } catch(e) {} }); }); req.on("error", () => {}); req.end(); } fetchGoldPrice(); setInterval(fetchGoldPrice, 30000); // ─── Account helper ────────────────────────────────────────── function getAccount(id, label) { if (!accounts[id]) { accounts[id] = { id, label:label||id, trades:[], openTrades:[], totalProfit:0, wins:0, losses:0, balance:0, equity:0, currency:"USD", server:"", online:false, lastSeen:null }; console.log(`[+] Account: ${label||id}`); broadcast({ type:"new_account", accountId:id, label:label||id }); } return accounts[id]; } // ─── HTTP server ───────────────────────────────────────────── const server = http.createServer((req, res) => { res.setHeader("Access-Control-Allow-Origin", "*"); res.setHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS"); res.setHeader("Access-Control-Allow-Headers", "Content-Type"); res.setHeader("ngrok-skip-browser-warning", "true"); if (req.method === "OPTIONS") { res.writeHead(204); res.end(); return; } if (req.method === "POST" && req.url === "/trade") { let body = ""; req.on("data", c => body += c); req.on("end", () => { try { handleTradeEvent(JSON.parse(body)); res.writeHead(200); res.end('{"ok":true}'); } catch(e) { res.writeHead(400); res.end("Bad JSON"); } }); return; } if (req.method === "POST" && req.url === "/contact") { let body = ""; req.on("data", c => body += c); req.on("end", () => { try { const d = JSON.parse(body); const contact = { name:d.name||"Visitor", whatsapp:d.whatsapp||"", time:new Date().toLocaleString("en-AE",{timeZone:"Asia/Dubai"}), ip:req.headers["x-forwarded-for"]||"unknown" }; contacts.push(contact); console.log(`[📱] NEW LEAD: ${contact.name} — wa:+${contact.whatsapp}`); broadcast({ type:"new_contact", contact }); res.writeHead(200); res.end('{"ok":true}'); } catch(e) { res.writeHead(400); res.end("err"); } }); return; } if (req.method === "GET" && req.url === "/contacts") { res.writeHead(200, {"Content-Type":"application/json"}); res.end(JSON.stringify({ total:contacts.length, contacts:[...contacts].reverse() })); return; } if (req.method === "GET" && req.url === "/status") { res.writeHead(200, {"Content-Type":"application/json"}); res.end(JSON.stringify({ status:"online", goldPrice, contacts:contacts.length, accounts:Object.keys(accounts).map(id=>({ id, label:accounts[id].label, trades:accounts[id].trades.length, profit:accounts[id].totalProfit, online:accounts[id].online, balance:accounts[id].balance })) })); return; } res.writeHead(404); res.end("Not found"); }); // ─── WebSocket ─────────────────────────────────────────────── const wss = new WebSocket.Server({ server }); wss.on("connection", ws => { console.log("[WS] Dashboard connected"); ws.send(JSON.stringify({ type:"init", accounts:buildSummary(), goldPrice })); ws.on("close", () => console.log("[WS] Dashboard disconnected")); }); function broadcast(data) { const msg = JSON.stringify(data); wss.clients.forEach(c => { if (c.readyState === WebSocket.OPEN) c.send(msg); }); } // ─── Trade Events ───────────────────────────────────────────── function handleTradeEvent(data) { const acc = getAccount(data.accountId, data.accountLabel); if (data.event === "heartbeat") { acc.online=true; acc.lastSeen=new Date(); acc.balance = data.balance || acc.balance; acc.equity = data.equity || acc.equity; acc.currency = data.currency || acc.currency; acc.server = data.server || acc.server; // Only use MT4 gold if Yahoo hasn't loaded yet if (!goldPrice && data.goldPrice && data.goldPrice > 100) { goldPrice = data.goldPrice; broadcast({ type:"gold_price", price:goldPrice }); } console.log(`[♥] ${acc.label} | $${acc.balance} | XAU:${goldPrice||'?'}`); broadcast({ type:"heartbeat", accountId:acc.id, label:acc.label, balance:acc.balance, equity:acc.equity, online:true, goldPrice }); return; } if (data.event === "open") { if (data.direction === "OTHER") return; // skip pending orders acc.openTrades = acc.openTrades.filter(t => t.ticket !== data.ticket); const trade = { ticket:data.ticket, symbol:data.symbol, direction:data.direction, entry:data.entry, lots:data.lots, sl:data.sl, tp:data.tp, time:data.time, comment:data.comment||"" }; acc.openTrades.push(trade); console.log(`[→] OPEN ${acc.label}: ${data.direction} ${data.symbol} @ ${data.entry}`); broadcast({ type:"trade_open", accountId:acc.id, label:acc.label, trade }); return; } if (data.event === "close") { acc.openTrades = acc.openTrades.filter(t => t.ticket !== data.ticket); // ✅ KEY FIX: parse real MT4 close date const closeDate = parseMT4Date(data.closeTime); const trade = { ticket:data.ticket, symbol:data.symbol, direction:data.direction, entry:data.entry, exit:data.exit, profit:data.profit, lots:data.lots, win:data.win, comment:data.comment||"", openTime:data.openTime, closeTime:data.closeTime, date: closeDate.toISOString() // ← real date from MT4! }; // Prevent duplicates (same ticket sent twice) if (acc.trades.some(t => t.ticket === data.ticket && t.profit === data.profit)) return; acc.trades.push(trade); acc.totalProfit = +(acc.totalProfit + data.profit).toFixed(2); if (data.win) acc.wins++; else acc.losses++; const winRate = acc.trades.length > 0 ? +((acc.wins/acc.trades.length)*100).toFixed(1) : 0; acc.winRate = winRate; console.log(`[✓] CLOSE ${acc.label}: ${data.symbol} P/L:$${data.profit} | Total:$${acc.totalProfit}`); broadcast({ type:"trade_close", accountId:acc.id, label:acc.label, trade, stats:{ totalProfit:acc.totalProfit, wins:acc.wins, losses:acc.losses, total:acc.trades.length, winRate } }); return; } } // ─── Build init summary ────────────────────────────────────── function buildSummary() { return Object.values(accounts).map(acc => ({ id:acc.id, label:acc.label, trades:acc.trades, openTrades:acc.openTrades, totalProfit:acc.totalProfit, wins:acc.wins, losses:acc.losses, balance:acc.balance, equity:acc.equity, currency:acc.currency, online:acc.online, lastSeen:acc.lastSeen, winRate: acc.trades.length > 0 ? +((acc.wins/acc.trades.length)*100).toFixed(1) : 0 })); } // ─── Offline detection ─────────────────────────────────────── setInterval(() => { const now = Date.now(); Object.values(accounts).forEach(acc => { if (acc.online && acc.lastSeen && (now - new Date(acc.lastSeen).getTime()) > 120000) { acc.online = false; console.log(`[!] ${acc.label} went offline`); broadcast({ type:"account_offline", accountId:acc.id }); } }); }, 30000); // ─── Start ─────────────────────────────────────────────────── server.listen(PORT, () => { console.log("╔══════════════════════════════════════╗"); console.log("║ SalmanFX Server v3 — RUNNING 🟢 ║"); console.log(`║ http://localhost:${PORT} ║`); console.log("║ GET /status → server health ║"); console.log("║ GET /contacts → see all leads 📱 ║"); console.log("║ Gold: Yahoo Finance auto 🪙 ║"); console.log("╚══════════════════════════════════════╝"); });
